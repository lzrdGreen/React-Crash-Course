Lesson 6: State Management Beyond useState

Objective

Manage complex state logic and share state between components effectively.

Topics Covered

  • Managing complex state with useReducer.
  • Sharing state using the Context API.
  • When and why to use these tools over useState.

Hands-On

Create a simple app to manage tasks using useReducer and share state across components using Context.

Example Code

  
    import { useReducer, createContext, useContext } from "react";

    const TaskContext = createContext();

    const taskReducer = (state, action) => {
      switch (action.type) {
        case "ADD_TASK":
          return [...state, { id: Date.now(), text: action.payload }];
        case "REMOVE_TASK":
          return state.filter(task => task.id !== action.payload);
        default:
          return state;
      }
    };

    function TaskProvider({ children }) {
      const [tasks, dispatch] = useReducer(taskReducer, []);
      return (
        <TaskContext.Provider value={{ tasks, dispatch }}>
          {children}
        </TaskContext.Provider>
      );
    }

    function TaskList() {
      const { tasks, dispatch } = useContext(TaskContext);

      return (
        <div>
          <ul>
            {tasks.map(task => (
              <li key={task.id}>
                {task.text}
                <button onClick={() => dispatch({ type: "REMOVE_TASK", payload: task.id })}>
                  Remove
                </button>
              </li>
            ))}
          </ul>
        </div>
      );
    }

    function TaskInput() {
      const { dispatch } = useContext(TaskContext);
      const addTask = () => {
        const task = prompt("Enter a task:");
        if (task) {
          dispatch({ type: "ADD_TASK", payload: task });
        }
      };

      return <button onClick={addTask}>Add Task</button>;
    }

    function App() {
      return (
        <TaskProvider>
          <TaskInput />
          <TaskList />
        </TaskProvider>
      );
    }

    export default App;
